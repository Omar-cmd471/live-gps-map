<head>
  <title>Live GPS Map</title>
  <meta charset="utf-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <!-- Leaflet.js CDN -->
  <link rel="stylesheet" href="https://unpkg.com/leaflet@1.7.1/dist/leaflet.css" />
  <script src="https://unpkg.com/leaflet@1.7.1/dist/leaflet.js"></script>
  <style>
    #map {
      height: 100vh;
      width: 100%;
    }
  </style>
</head>
<body>
  <div id="map"></div>

  <script>
    const channelID = 2920529;
    const readAPIKey = 'EO08IPO7PM9OCRGG';
    const updateInterval = 60000; // 60 ثانية

    let map = L.map('map').setView([33.3, 44.4], 17); // موقع مبدئي

    // إعداد الخريطة
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '© OpenStreetMap'
    }).addTo(map);

    let marker = L.marker([0, 0]).addTo(map);

    function fetchGPS() {
      const url = `https://api.thingspeak.com/channels/${channelID}/feeds.json?api_key=${readAPIKey}&results=1`;
      fetch(url)
        .then(response => response.json())
        .then(data => {
          let feed = data.feeds[0];
          if (feed) {
            let lat = parseFloat(feed.field1);
            let lng = parseFloat(feed.field2);
            if (!isNaN(lat) && !isNaN(lng)) {
              marker.setLatLng([lat, lng]);
              map.setView([lat, lng], 18);
            }
          }
        })
        .catch(err => console.error("Error fetching data:", err));
    }

    // تحديث مباشر
    fetchGPS();
    setInterval(fetchGPS, updateInterval);
  </script>
</body>
